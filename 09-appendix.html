[root@localhost a]# vim setup_redis.yml
- hosts: webservers
  vars:
    - PORT: 6379

  tasks:
    - name: "安装yum源"
      yum: name=epel-release state=present

    - name: "安装redis"
      yum:

    - name: "下发redis的配置文件"
#copy 或者 template  下发/etc/redis.conf

      notify: restart_redis

    - name: "启动redis"
      shell: redis-server /etc/redis.conf &

    - name: "设置redis开机自启"
      service:
        name: redis
        enabled: yes

  handlers:
    - name: restart_redis
      shell:  redis-server /etc/redis.conf &
